# .vimrc

inoremap jk <Esc>
inoremap kj <Esc>
inoremap JK <Esc>
inoremap KJ <Esc>

set number
set relativenumber
set tabstop=4 softtabstop=4 shiftwidth=4 expandtab
set autoindent
set smartindent

set mouse=a
map <ScrollWheelUp> <C-Y>
map <ScrollWheelDown> <C-E>

syntax on
set background=dark
colorscheme slate
